81 const long int innerToOuterDiff = gl_nVoxel[X] / 2 - sideLength / 2;
82 const long int rightSide = innerToOuterDiff + sideLength;

85 // Be careful, with GCC the OpenMP collapse doesn't work if you declare the indexes as unsigned short!!
86#pragma omp parallel for collapse(3) default(none) shared(f, nOfSlices, gl_nVoxel, offset, sideLength, innerToOuterDiff, rightSide)
87 for (unsigned n = 0 ; n < nOfSlices; n++) {
88 for (unsigned i = 0; i < gl_nVoxel[Z]; i++) {
89 for (unsigned j = 0; j < gl_nVoxel[X]; j++) {

142void generateCubeWithSphereSlice(double *f, unsigned short nOfSlices, unsigned short offset, const unsigned sideLength)
143{
144 const long int innerToOuterDiff = gl_nVoxel[X] / 2 - sideLength / 2;
145 const long int rightSide = innerToOuterDiff + sideLength;

151 for (unsigned n = 0 ; n < nOfSlices; n++) {
152 for (unsigned i = 0; i < gl_nVoxel[Z]; i++) {
153 for (unsigned j = 0; j < gl_nVoxel[X]; j++) {
154 f[(unsigned) gl_nVoxel[Z] * i + j + (unsigned) n * gl_nVoxel[X] * gl_nVoxel[Z]] = 0.0;
155 if (i >= innerToOuterDiff && i <= rightSide && j >= innerToOuterDiff && j <= rightSide
156 && n + offset >= innerToOuterDiff && n + offset <= gl_nVoxel[Y] - innerToOuterDiff) {
157 // Voxel position is inside the cubic object
158 Point temp;
159 temp.y = -(double) gl_objectSideLength / 2.0 + gl_voxelYDim / 2.0 + (n + offset) * gl_voxelYDim;
160 temp.x = -(double) gl_objectSideLength / 2.0 + gl_voxelXDim / 2.0 + j * gl_voxelXDim;
161 temp.z = -(double) gl_objectSideLength / 2.0 + gl_voxelZDim / 2.0 + i * gl_voxelZDim;
162 const double distance = sqrt(pow(temp.x - sphereCenter.x, 2) + pow(temp.y - sphereCenter.y, 2) + pow(temp.z - sphereCenter.z, 2));
163
164 if (distance > sideLength * gl_voxelXDim / 6.0) {
165 // Voxel position is outside the spherical cavity
166 f[(unsigned) gl_nVoxel[Z] * i + j + (unsigned) n * gl_nVoxel[X] * gl_nVoxel[Z]] = 1.0;

254int main(int argc, char *argv[])
255{
256 int objectType = 0; // represents the object type chosen
257 // int n = DEFAULT_WORK_SIZE; // number of voxels along the detector's
258
259 if (argc > 4 || argc < 2) {
260 fprintf(stderr,
261 "Usage:\n%s DEST [OBJECT] [PIXELS]\nWhere:\n"
262 " - DEST: is the output file name.\n"
263 " - [OBJECT]: can be: '%s' (default), '%s' or '%s'.\n"
264 " - [PIXELS]: is the number of pixels per side of the detector, every other parameter is set based to its"
265 " value, if no value is given, default values are used.\n",
266 argv[0],
267 CUBE,
268 SPHERE,
269 CUBE_WITH_SPHERICAL_HOLE);
270 return EXIT_FAILURE;
271 }
272 char *fileName = argv[1];
273 if (argc > 2) {
274 if (strcmp(argv[2], CUBE_WITH_SPHERICAL_HOLE) == 0) {
275 objectType = 1;
276 } else if (strcmp(argv[2], SPHERE) == 0) {
277 objectType = 2;
278 } else if (strcmp(argv[2], CUBE) != 0) {
279 fprintf(stderr,
280 "Usage:\n%s DEST [OBJECT] [PIXELS]\nWhere:\n"
281 " - [OBJECT]: can be: '%s' (default), '%s' or '%s', nothing else is accepted.\n",
282 argv[0],
283 CUBE,
284 SPHERE,
285 CUBE_WITH_SPHERICAL_HOLE);
286 return EXIT_FAILURE;
287 }
288 if (argc > 3) {
289 const int n = atoi(argv[3]);
290 gl_objectSideLength = n * gl_voxelXDim * ((double) OBJECT_SIDE_LENGTH / (VOXEL_X_DIM * N_PIXEL_ALONG_SIDE));
291 gl_detectorSideLength = n * gl_pixelDim;
292 gl_distanceObjectDetector = 1.5 * gl_objectSideLength;
293 gl_distanceObjectSource = 6 * gl_objectSideLength;
294 gl_nVoxel[X] = gl_objectSideLength / gl_voxelXDim;
295 gl_nVoxel[Y] = gl_objectSideLength / gl_voxelYDim;
296 gl_nVoxel[Z] = gl_objectSideLength / gl_voxelZDim;
297 gl_nPlanes[X] = gl_nVoxel[X] + 1;
298 gl_nPlanes[Y] = gl_nVoxel[Y] + 1;
299 gl_nPlanes[Z] = gl_nVoxel[Z] + 1;
300 }
301 }
302
303 // Array containing the coefficients of each voxel
304 double *f = (double *) malloc(sizeof(double) * gl_nVoxel[X] * OBJ_BUFFER * gl_nVoxel[Z]);
305
306 FILE *filePointer = fopen(fileName, "wb");
307 if (!filePointer) {
308 fprintf(stderr, "Unable to open file '%s' in write binary mode!\n", fileName);
309 return EXIT_FAILURE;
310 }
311
312 // Write the voxel grid dimensions on file
313 const size_t headerLength = writeSetUp(filePointer);
314 if (headerLength < 0) {
315 fprintf(stderr, "Unable to write on file '%s'!\n", fileName);
316 return EXIT_FAILURE;
317 }
318
319 // Iterates over each object subsection which size is limited along the y coordinate by OBJ_BUFFER
320 for (unsigned short slice = 0; slice < gl_nVoxel[Y]; slice += OBJ_BUFFER) {
321 unsigned short nOfSlices;
322
323 if (gl_nVoxel[Y] - slice < OBJ_BUFFER) {
324 nOfSlices = gl_nVoxel[Y] - slice;
325 } else {
326 nOfSlices = OBJ_BUFFER;
327 }
328
329 // Generates object subsection
330 switch (objectType) {
331 case 1:
332 generateCubeWithSphereSlice(f, nOfSlices, slice, gl_nVoxel[X]);
333 break;
334 case 2:
335 generateSphereSlice(f, nOfSlices, slice, gl_objectSideLength / 2);
336 break;
337 default:
338 generateCubeSlice(f, nOfSlices, slice, gl_nVoxel[X]);
339 break;
340 }
341
342 if (!fwrite(f, sizeof(double), (size_t) gl_nVoxel[X] * gl_nVoxel[Z] * nOfSlices, filePointer)) {
343 fprintf(stderr, "Unable to write on file '%s'!\n", fileName);
344 return EXIT_FAILURE;
345 }
346 }
347
348 printf("Output file details:\n");
349 printSizeMaxGB(" Voxel model size:", sizeof(double) * gl_nVoxel[X] * gl_nVoxel[Z] * gl_nVoxel[Y]);
350 printf(" Image type: %lu bit real\n", sizeof(double) * 8);
351 printf(" Image width: %d pixels\n", gl_nVoxel[X]);
352 printf(" Image height: %d pixels\n", gl_nVoxel[Z]);
353 printf(" Offset to first image: %lu Bytes\n", headerLength);
354 printf(" Number of images: %d\n", gl_nVoxel[Y]);
355 printf(" Gap between images: 0 bytes\n");
356
357 fclose(filePointer);
358 free(f);
359
360 return EXIT_SUCCESS;
361}
