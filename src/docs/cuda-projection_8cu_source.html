243__device__ void getAllIntersections(const double source, const double pixel, const Ranges planeIndexesRanges, double *const a, const Axis axis)
244{
245 int start = 0, end = 0;
246 double d;
247
248 start = planeIndexesRanges.minIndx;
249 end = planeIndexesRanges.maxIndx;
250 if (end > start) { // Avoids management of invalid array
251 assert(end - start <= MAX_PLANES);
252 double plane[MAX_PLANES];
253 if (axis == X) {
254 plane[0] = getXPlane(start);
255 d = d_voxelXDim;
256 if (pixel - source < 0) {
257 plane[0] = getXPlane(end);
258 d = -(double) d_voxelXDim;
259 }
260 } else if (axis == Y) {
261 plane[0] = getYPlane(start);
262 d = d_voxelYDim;
263 if (pixel - source < 0) {
264 plane[0] = getYPlane(end);
265 d = -(double) d_voxelYDim;
266 }
267 } else /* if (axis == Z) */ {
268 plane[0] = getZPlane(start);
269 d = d_voxelZDim;
270 if (pixel - source < 0) {
271 plane[0] = getZPlane(end);
272 d = -(double) d_voxelZDim;
273 }
274 }
275
276 for (unsigned short i = 1; i < end - start; i++) {
277 plane[i] = plane[i - 1] + d;
278 }
279 getIntersection(source, pixel, plane, end - start, a);
280 }
281}

295__device__ Ranges getRangeOfIndex(const double source, const double pixel, const char isParallel, const double aMin, const double aMax, const Axis axis)
296{
297 Ranges idxs;
298 double firstPlane, lastPlane;
299 unsigned short voxelDim;
300
301 if (axis == X) {
302 voxelDim = d_voxelXDim;
303 firstPlane = getXPlane(0);
304 lastPlane = getXPlane(d_nPlanes[X] - 1);
305 } else if (axis == Y) {
306 voxelDim = d_voxelYDim;
307 firstPlane = getYPlane(0);
308 lastPlane = getYPlane(d_nPlanes[Y] - 1);
309 } else /* if (axis == Z) */ {
310 voxelDim = d_voxelZDim;
311 firstPlane = getZPlane(0);
312 lastPlane = getZPlane(d_nPlanes[Z] - 1);
313 }
314
315 // Gets range of indexes of XZ parallel planes
316 if (isParallel != Y) {
317 if (pixel - source >= 0) {
318 idxs.minIndx = d_nPlanes[axis] - ceil((lastPlane - aMin * (pixel - source) - source) / voxelDim);
319 idxs.maxIndx = 1 + floor((aMax * (pixel - source) + source - firstPlane) / voxelDim);
320 } else {
321 idxs.minIndx = d_nPlanes[axis] - ceil((lastPlane - aMax * (pixel - source) - source) / voxelDim);
322 idxs.maxIndx = floor((aMin * (pixel - source) + source - firstPlane) / voxelDim);
323 }
324 } else {
325 idxs.minIndx = 0;
326 idxs.maxIndx = 0;
327 }
328 return idxs;
329}

342__device__ unsigned short merge(const double *const a, const double *const b, const unsigned short lenA, const unsigned short lenB, double *const c)
343{
344 unsigned short i = 0, j = 0, k = 0;
345 while (j < lenA && k < lenB) {
346 if (a[j] < b[k]) {
347 c[i] = a[j];
348 j++;
349 } else {
350 c[i] = b[k];
351 k++;
352 }
353 i++;
354 }
355 while (j < lenA) {
356 c[i] = a[j];
357 i++;
358 j++;
359 }
360 while (k < lenB) {
361 c[i] = b[k];
362 i++;
363 k++;
364 }
365 return lenA + lenB;
366}

420__device__ Point getPixel(const double *const sinTable, const double *const cosTable, const unsigned r, const unsigned c, const unsigned short index)
421{
422 Point pixel;
423 const double sinAngle = sinTable[index];
424 const double cosAngle = cosTable[index];
425 const double elementOffset = d_detectorSideLength / 2 - d_pixelDim / 2;
426
427 pixel.x = -(double) d_distanceObjectDetector * sinAngle + cosAngle * (-elementOffset + (double) d_pixelDim * c);
428 pixel.y = -(double) d_distanceObjectDetector * cosAngle - sinAngle * (-elementOffset + (double) d_pixelDim * c);
429 pixel.z = -elementOffset + (double) d_pixelDim * r;
430
431 return pixel;
432}

486__device__ double computeAbsorption(const unsigned short slice, const Point source, const Point pixel, const double *const a, const unsigned short lenA, const double *const f)
487{
488 double g = 0.0;
489
490 if (lenA > 0) { // Avoids overflow on unsigned value
491 const double deltaX = pixel.x - source.x;
492 const double deltaY = pixel.y - source.y;
493 const double deltaZ = pixel.z - source.z;
494 const double d12 = sqrt(pow(deltaX, 2) + pow(deltaY, 2) + pow(deltaZ, 2));
495 for (unsigned short i = 0; i < lenA - 1; i++) {
496 const double aMid = (a[i + 1] + a[i]) / 2;
497 const unsigned short x = min((int) ((source.x + aMid * deltaX - getXPlane(0)) / d_voxelXDim), d_nVoxel[X] - 1);
498 const unsigned short y = min((int) ((source.y + aMid * deltaY - getYPlane(slice)) / d_voxelYDim), min(d_nVoxel[Y] - 1, d_yVoxels - 1));
499 const unsigned short z = min((int) ((source.z + aMid * deltaZ - getZPlane(0)) / d_voxelZDim), d_nVoxel[Z] - 1);
500
501 // In a 3D matrix it would be: f[x][z][y]
502 // d12 * (a[i + 1] - a[i] = segment length
503 g += f[x + (unsigned) z*d_nVoxel[Z] + (unsigned) y*d_nVoxel[X]*d_nVoxel[Z]] * d12 * (a[i + 1] - a[i]);
504 }
505 }
506 return g;
507}

516__device__ __forceinline__ double atomicMinDouble(double *const addr, const double value) {
517 unsigned long long *addr_as_ull = (unsigned long long *) addr;
518 unsigned long long old = *addr_as_ull, assumed;
519
520 do {
521 assumed = old;
522 old = atomicCAS(addr_as_ull, assumed, __double_as_longlong(fmin(__longlong_as_double(assumed), value)));
523 } while (assumed != old);
524
525 return __longlong_as_double(old);
526}

558__global__ void computeProjections(const unsigned short slice, const unsigned short nTheta, const unsigned nSidePixels, const double *const f, double *const g, const char isFirst) {
559 __shared__ double l_gMins[BLKDIM];
560 __shared__ double l_gMaxs[BLKDIM];
561 const unsigned l_index = threadIdx.x + threadIdx.y * blockDim.x;
562 const unsigned r = threadIdx.y + blockIdx.y * blockDim.y;
563 const unsigned c = threadIdx.x + blockIdx.x * blockDim.x;
564 l_gMins[l_index] = INFINITY;
565 l_gMaxs[l_index] = -INFINITY;
566
567 if (r < nSidePixels && c < nSidePixels) {
568 double a[3][2];
569
570 if (isFirst) {
571 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
572 g[positionIndex * nSidePixels * nSidePixels + r * nSidePixels + c] = 0.0;
573 }
574 }
575 // Iterates over each source
576 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
577 const Point source = getSource(d_gl_sinTable, d_gl_cosTable, positionIndex);
578
579 // Computes the attenuation over a single pixel of the detector
580 const Point pixel = getPixel(d_gl_sinTable, d_gl_cosTable, r, c, positionIndex);
581
582 // Computes Min-Max parametric values
583 double aMin, aMax;
584 double sidesPlanes[2];
585 char isParallel = -1;
586 getSidesXPlanes(sidesPlanes);
587 if (!getIntersection(source.x, pixel.x, sidesPlanes, 2, &a[X][0])) {
588 isParallel = X;
589 }
590 getSidesYPlanes(sidesPlanes, slice);
591 if (!getIntersection(source.y, pixel.y, sidesPlanes, 2, &a[Y][0])) {
592 isParallel = Y;
593 }
594 getSidesZPlanes(sidesPlanes);
595 if (!getIntersection(source.z, pixel.z, sidesPlanes, 2, &a[Z][0])) {
596 isParallel = Z;
597 }
598
599 aMin = getAMin(a, isParallel);
600 aMax = getAMax(a, isParallel);
601
602 if (aMin < aMax) {
603 // Computes Min-Max plane indexes
604 Ranges indices[3];
605 indices[X] = getRangeOfIndex(source.x, pixel.x, isParallel, aMin, aMax, X);
606 indices[Y] = getRangeOfIndex(source.y, pixel.y, isParallel, aMin, aMax, Y);
607 indices[Z] = getRangeOfIndex(source.z, pixel.z, isParallel, aMin, aMax, Z);
608
609 // Computes lengths of the arrays containing parametric value of the intersection with each set of parallel planes
610 const unsigned short lenX = max(0, indices[X].maxIndx - indices[X].minIndx);
611 const unsigned short lenY = max(0, indices[Y].maxIndx - indices[Y].minIndx);
612 const unsigned short lenZ = max(0, indices[Z].maxIndx - indices[Z].minIndx);
613
614 // Computes ray-planes intersection Nx + Ny + Nz
615 double aX[MAX_PLANES];
616 double aY[MAX_PLANES];
617 double aZ[MAX_PLANES];
618 getAllIntersections(source.x, pixel.x, indices[X], aX, X);
619 getAllIntersections(source.y, pixel.y, indices[Y], aY, Y);
620 getAllIntersections(source.z, pixel.z, indices[Z], aZ, Z);
621
622 // Computes segments Nx + Ny + Nz
623 double aMerged[MAX_PLANES_x3];
624 const unsigned short lenA = merge3(aX, aY, aZ, lenX, lenY, lenZ, aMerged);
625
626 // Associates each segment to the respective voxel Nx + Ny + Nz
627 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + r * nSidePixels + c;
628 g[pixelIndex] += computeAbsorption(slice, source, pixel, aMerged, lenA, f);
629 l_gMins[l_index] = fmin(l_gMins[l_index], g[pixelIndex]);
630 l_gMaxs[l_index] = fmax(l_gMaxs[l_index], g[pixelIndex]);
631 }
632 }
633 }
634
635 unsigned b_size = blockDim.x / 2;
636 __syncthreads();
637 while (b_size > 0) {
638 if (l_index < b_size) {
639 if (l_gMins[l_index] > l_gMins[l_index + b_size]) {
640 l_gMins[l_index] = l_gMins[l_index + b_size];
641 }
642 if (l_gMaxs[l_index] < l_gMaxs[l_index + b_size]) {
643 l_gMaxs[l_index] = l_gMaxs[l_index + b_size];
644 }
645 }
646 b_size = b_size / 2;
647 __syncthreads();
648 }
649
650 if (l_index == 0) {
651 atomicMinDouble(&d_gMin, l_gMins[l_index]);
652 atomicMaxDouble(&d_gMax, l_gMaxs[l_index]);
653 }
654}

765 exit(EXIT_FAILURE);
766 }

768 }

770 *sizeF = size;
771 } else {
772 *sizeF = sizeof(double) * gl_nVoxel[X] * yVoxels * gl_nVoxel[Z];
773 }
774
775 cudaSafeCall(cudaMalloc((void **) &d_f, *sizeF));
776 cudaSafeCall(cudaMemcpyToSymbol(d_yVoxels, &yVoxels, sizeof(d_yVoxels)));
777#ifdef DEBUG
778 printSizeMaxGB("f", *sizeF, "GLOBAL");
779 printSizeMaxGB("g", sizeG, "GLOBAL");
780 printf("\n");
781#endif
782}

794void asyncComputeProjections(const unsigned short slice, double *f, const size_t sizeF, const unsigned short nTheta, const unsigned nSidePixels, const char isFirst)
795{
796#ifdef PRINT
797 printf("%.4lf> Copying f...\n", hpc_gettime());
798#endif
799 cudaSafeCall(cudaMemcpy(d_f, f, sizeF, cudaMemcpyHostToDevice));
800 static dim3 block(BLKDIM_STEP, BLKDIM_STEP);
801 static dim3 grid((nSidePixels + BLKDIM_STEP - 1) / BLKDIM_STEP, (nSidePixels + BLKDIM_STEP - 1) / BLKDIM_STEP);

870int main(int argc, char *argv[])
871{
872 if (argc < 2 || argc > 4) {
873 fprintf(stderr, "Usage: %s INPUT [OUTPUT] [Y_MAX_VOXELS]\n"
874 "- INPUT: The first parameter is the name of the input file.\n"
875 "- [OUTPUT]: The second parameter is the name of a .pgm file to store the output at.\n"
876 "- [Y_MAX_VOXELS]: The third parameter is the maximum number of voxels considered in the Y axis for each iteration.\n",
877 argv[0]);
878 return EXIT_FAILURE;
879 }
880 const char *const inputFileName = argv[1];
881 const char *outputFileName = NULL;
882 if (argc > 2) {
883 outputFileName = argv[2];
884 }
885
886 FILE *const inputFilePointer = fopen(inputFileName, "rb");
887 if (!inputFilePointer) {
888 fprintf(stderr, "Unable to open file '%s'!\n", inputFileName);
889 return EXIT_FAILURE;
890 }
891
892 if (readSetUP(inputFilePointer) == EXIT_FAILURE) {
893 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
894 return EXIT_FAILURE;
895 }
896 if (gl_nPlanes[X] > MAX_PLANES || gl_nPlanes[Y] > MAX_PLANES || gl_nPlanes[Z] > MAX_PLANES) {
897 fprintf(stderr, "The maximum number of planes per axis is %u planes!\n", MAX_PLANES);
898 return EXIT_FAILURE;
899 }
900 if (argc > 3) {
901 const int yMaxVoxels = atoi(argv[3]);
902 yVoxels = min(gl_nVoxel[Y], max(yVoxels, 1));
903 }
904 // Number of angular positions
905 double partialTime = hpc_gettime();
906 const unsigned short nTheta = gl_angularTrajectory / gl_positionsAngularDistance + 1;
907 if (nTheta > MAX_TABLES_SIZE) {
908 fprintf(stderr, "Number of positions required %u is too large, max %lu!\n", nTheta, MAX_TABLES_SIZE);
909 exit(EXIT_FAILURE);
910 }
911 const unsigned nSidePixels = gl_detectorSideLength / gl_pixelDim;
912 // Size of the array containing the computed attenuation detected in each pixel of the detector
913 const size_t sizeG = sizeof(double) * nSidePixels * nSidePixels * nTheta;
914 // Minimum and maximum attenuation computed
915 double gMinValue = INFINITY, gMaxValue = -INFINITY;
916 size_t sizeF;
917 initEnvironment(&sizeF, sizeG, nTheta, nSidePixels, &gMinValue, &gMaxValue);
918 // Array containing the coefficients of each voxel
919 double *const f = (double *) malloc(sizeF);
920 double totalTime = hpc_gettime() - partialTime;
921
922 // Iterates over object subsections
923 for (unsigned short slice = 0; slice < gl_nVoxel[Y]; slice += yVoxels) {
924 unsigned short nOfSlices;
925
926 if (gl_nVoxel[Y] - slice < yVoxels) {
927 nOfSlices = gl_nVoxel[Y] - slice;
928 } else {
929 nOfSlices = yVoxels;
930 }
931
932 // Read voxels coefficients
933#ifdef PRINT
934 printf("%.4lf> Reading f...\n", hpc_gettime());
935#endif
936 if (!fread(f, sizeof(double), (size_t) gl_nVoxel[X] * nOfSlices * gl_nVoxel[Z], inputFilePointer)) {
937 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
938 free(f);
939 termEnvironment();
940 return EXIT_FAILURE;
941 }
942
943#ifdef PRINT
944 static unsigned short it = 0;
945 printf("%.4lf> IT %u of size %hu\n", hpc_gettime(), ++it, nOfSlices);
946#endif
947 // Computes subsection projection
948 partialTime = hpc_gettime();
949 asyncComputeProjections(slice, f, sizeF, nTheta, nSidePixels, !slice);
950 totalTime += hpc_gettime() - partialTime;
951 }
952 fclose(inputFilePointer);
953 partialTime = hpc_gettime();
954 free(f);
955 // Array containing the computed attenuation detected in each pixel of the detector
956 double *const g = (double *) malloc(sizeG);
957 termEnvironment(g, sizeG, &gMinValue, &gMaxValue);
958 totalTime += hpc_gettime() - partialTime;
959 printf("Execution time (s) %.2f\n", totalTime);
960
961 if (outputFileName != NULL) {
962 FILE *const outputFilePointer = fopen(outputFileName, "w");
963 if (!outputFileName) {
964 fprintf(stderr, "Unable to open file '%s'!\n", outputFileName);
965 free(g);
966 return EXIT_FAILURE;
967 }
968 // Iterates over each attenuation value computed, prints a value between [0-255]
969 fprintf(outputFilePointer, "P2\n%d %d\n255", nSidePixels, nSidePixels * nTheta);
970 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
971 const double angle = -(double) gl_angularTrajectory / 2 + (double) positionIndex * gl_positionsAngularDistance;
972 fprintf(outputFilePointer, "\n#%lf", angle);
973 for (unsigned i = 0; i < nSidePixels; i++) {
974 fprintf(outputFilePointer, "\n");
975 for (unsigned j = 0; j < nSidePixels; j++) {
976 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + i * nSidePixels + j;
977 const int color = (g[pixelIndex] - gMinValue) * 255 / (gMaxValue - gMinValue);
978 fprintf(outputFilePointer, "%d ", color);
979 }
980 }
981 }
982 fclose(outputFilePointer);
983 }
984 free(g);
985
986 return EXIT_SUCCESS;
987}
