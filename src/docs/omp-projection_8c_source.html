230void getAllIntersections(const double source, const double pixel, const Ranges planeIndexesRanges, double *const a, const Axis axis)
231{
232 int start = 0, end = 0;
233 double d;
234
235 start = planeIndexesRanges.minIndx;
236 end = planeIndexesRanges.maxIndx;
237 if (end > start) { // Avoids management of invalid array
238 double plane[end - start];
239 if (axis == X) {
240 plane[0] = getXPlane(start);
241 d = gl_voxelXDim;
242 if (pixel - source < 0) {
243 plane[0] = getXPlane(end);
244 d = -gl_voxelXDim;
245 }
246 } else if (axis == Y) {
247 plane[0] = getYPlane(start);
248 d = gl_voxelYDim;
249 if (pixel - source < 0) {
250 plane[0] = getYPlane(end);
251 d = -gl_voxelYDim;
252 }
253 } else /* if (axis == Z) */ {
254 plane[0] = getZPlane(start);
255 d = gl_voxelZDim;
256 if (pixel - source < 0) {
257 plane[0] = getZPlane(end);
258 d = -gl_voxelZDim;
259 }
260 }
261
262 for (unsigned short i = 1; i < end - start; i++) {
263 plane[i] = plane[i - 1] + d;
264 }
265 getIntersection(source, pixel, plane, end - start, a);
266 }

280Ranges getRangeOfIndex(const double source, const double pixel, const char isParallel, const double aMin, const double aMax, const Axis axis)
281{
282 Ranges idxs;
283 double firstPlane, lastPlane;
284 unsigned short voxelDim;
285
286 if (axis == X) {
287 voxelDim = gl_voxelXDim;
288 firstPlane = getXPlane(0);
289 lastPlane = getXPlane(gl_nPlanes[X] - 1);
290 } else if (axis == Y) {
291 voxelDim = gl_voxelYDim;
292 firstPlane = getYPlane(0);
293 lastPlane = getYPlane(gl_nPlanes[Y] - 1);
294 } else /* if (axis == Z) */ {
295 voxelDim = gl_voxelZDim;
296 firstPlane = getZPlane(0);
297 lastPlane = getZPlane(gl_nPlanes[Z] - 1);
298 }
299
300 // Gets range of indexes of XZ parallel planes
301 if (isParallel != Y) {
302 if (pixel - source >= 0) {
303 idxs.minIndx = gl_nPlanes[axis] - ceil((lastPlane - aMin * (pixel - source) - source) / voxelDim);
304 idxs.maxIndx = 1 + floor((aMax * (pixel - source) + source - firstPlane) / voxelDim);
305 } else {
306 idxs.minIndx = gl_nPlanes[axis] - ceil((lastPlane - aMax * (pixel - source) - source) / voxelDim);
307 idxs.maxIndx = floor((aMin * (pixel - source) + source - firstPlane) / voxelDim);
308 }
309 } else {
310 idxs.minIndx = 0;
311 idxs.maxIndx = 0;
312 }
313 return idxs;
314}

326int merge(const double *const a, const double *const b, const unsigned short lenA, const unsigned short lenB, double *const c)
327{
328 int i = 0, j = 0, k = 0;
329 while (j < lenA && k < lenB) {
330 if (a[j] < b[k]) {
331 c[i] = a[j];
332 j++;
333 } else {
334 c[i] = b[k];
335 k++;
336 }
337 i++;
338 }
339 while (j < lenA) {
340 c[i] = a[j];
341 i++;
342 j++;
343 }
344 while (k < lenB) {
345 c[i] = b[k];
346 i++;
347 k++;
348 }
349 return lenA + lenB;
350}

400Point getPixel(const double *const sinTable, const double *const cosTable, const unsigned r, const unsigned c, const unsigned short index)
401{
402 Point pixel;
403 const double sinAngle = sinTable[index];
404 const double cosAngle = cosTable[index];
405 const double elementOffset = gl_detectorSideLength / 2 - gl_pixelDim / 2;
406
407 pixel.x = -(double) gl_distanceObjectDetector * sinAngle + cosAngle * (-elementOffset + (double) gl_pixelDim * c);
408 pixel.y = -(double) gl_distanceObjectDetector * cosAngle - sinAngle * (-elementOffset + (double) gl_pixelDim * c);
409 pixel.z = -elementOffset + (double) gl_pixelDim * r;

463double computeAbsorption(const unsigned short slice, const Point source, const Point pixel, const double *const a, const unsigned short lenA, const double *const f)
464{
465 double g = 0.0;
466
467 if (lenA > 0) { // Avoids overflow on unsigned value
468 const double deltaX = pixel.x - source.x;
469 const double deltaY = pixel.y - source.y;
470 const double deltaZ = pixel.z - source.z;
471 const double d12 = sqrt(pow(deltaX, 2) + pow(deltaY, 2) + pow(deltaZ, 2));
472 for (unsigned short i = 0; i < lenA - 1; i++) {
473 const double aMid = (a[i + 1] + a[i]) / 2;
474 const unsigned short x = min((source.x + aMid * deltaX - getXPlane(0)) / gl_voxelXDim, gl_nVoxel[X] - 1);
475 const unsigned short y = min3((source.y + aMid * deltaY - getYPlane(slice)) / gl_voxelYDim, gl_nVoxel[Y] - 1, yVoxels - 1);
476 const unsigned short z = min((source.z + aMid * deltaZ - getZPlane(0)) / gl_voxelZDim, gl_nVoxel[Z] - 1);
477
478 // In a 3D matrix it would be: f[x][z][y]
479 // d12 * (a[i + 1] - a[i] = segment length
480 g += f[x + (unsigned) z*gl_nVoxel[Z] + (unsigned) y*gl_nVoxel[X]*gl_nVoxel[Z]] * d12 * (a[i + 1] - a[i]);
481 }
482 }
483 return g;
484}

497void computeProjections(const unsigned short slice, double *f, double *g, double *gMin, double *gMax, const unsigned short nTheta, const unsigned nSidePixels)
498{
499 double l_gMin = INFINITY;
500 double l_gMax = -INFINITY;
501
502 // Iterates over each source
503 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
504 const Point source = getSource(gl_sinTable, gl_cosTable, positionIndex);
505
506 // Iterates over each pixel of the detector
507#pragma omp parallel for collapse(2) schedule(dynamic) default(none) shared(gl_sinTable, gl_cosTable, nSidePixels, positionIndex, source, slice, f, g, nTheta, gl_nVoxel, gl_nPlanes) reduction(min:l_gMin) reduction(max:l_gMax)
508 for (unsigned r = 0; r < nSidePixels; r++) {
509 for (unsigned c = 0; c < nSidePixels; c++) {
510 double a[3][2];
511
512 // Gets the pixel's center cartesian coordinates
513 const Point pixel = getPixel(gl_sinTable, gl_cosTable, r, c, positionIndex);
514
515 // Computes Min-Max parametric values
516 double aMin, aMax;
517 double sidesPlanes[2];
518 char isParallel = -1;
519 getSidesXPlanes(sidesPlanes);
520 if (!getIntersection(source.x, pixel.x, sidesPlanes, 2, &a[X][0])) {
521 isParallel = X;
522 }
523 getSidesYPlanes(sidesPlanes, slice);
524 if (!getIntersection(source.y, pixel.y, sidesPlanes, 2, &a[Y][0])) {
525 isParallel = Y;
526 }
527 getSidesZPlanes(sidesPlanes);
528 if (!getIntersection(source.z, pixel.z, sidesPlanes, 2, &a[Z][0])) {
529 isParallel = Z;
530 }
531
532 aMin = getAMin(a, isParallel);
533 aMax = getAMax(a, isParallel);
534
535 if (aMin < aMax) {
536 // Computes Min-Max plane indexes
537 Ranges indices[3];
538 indices[X] = getRangeOfIndex(source.x, pixel.x, isParallel, aMin, aMax, X);
539 indices[Y] = getRangeOfIndex(source.y, pixel.y, isParallel, aMin, aMax, Y);
540 indices[Z] = getRangeOfIndex(source.z, pixel.z, isParallel, aMin, aMax, Z);
541
542 // Computes lengths of the arrays containing parametric value of the intersection with each set of parallel planes
543 const unsigned short lenX = max(0, indices[X].maxIndx - indices[X].minIndx);
544 const unsigned short lenY = max(0, indices[Y].maxIndx - indices[Y].minIndx);
545 const unsigned short lenZ = max(0, indices[Z].maxIndx - indices[Z].minIndx);
546
547 // Computes ray-planes intersection Nx + Ny + Nz
548 double aX[gl_nPlanes[X]];
549 double aY[gl_nPlanes[Y]];
550 double aZ[gl_nPlanes[Z]];
551 getAllIntersections(source.x, pixel.x, indices[X], aX, X);
552 getAllIntersections(source.y, pixel.y, indices[Y], aY, Y);
553 getAllIntersections(source.z, pixel.z, indices[Z], aZ, Z);
554
555 // Computes segments Nx + Ny + Nz
556 double aMerged[gl_nPlanes[X] + gl_nPlanes[Y] + gl_nPlanes[Z]];
557 const unsigned short lenA = merge3(aX, aY, aZ, lenX, lenY, lenZ, aMerged);
558
559 // Associates each segment to the respective voxel Nx + Ny + Nz
560 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + r * nSidePixels + c;
561 g[pixelIndex] += computeAbsorption(slice, source, pixel, aMerged, lenA, f);
562 l_gMax = fmax(l_gMax, g[pixelIndex]);
563 l_gMin = fmin(l_gMin, g[pixelIndex]);
564 }
565 }
566 }
567 }
568 *gMax = l_gMax;
569 *gMin = l_gMin;
570}

654int main(int argc, char *argv[])
655{
656 if (argc < 2 || argc > 4) {
657 fprintf(stderr, "Usage: %s INPUT [OUTPUT] [Y_MAX_VOXELS]\n"
658 "- INPUT: The first parameter is the name of the input file.\n"
659 "- [OUTPUT]: The second parameter is the name of a .pgm file to store the output at.\n"
660 "- [Y_MAX_VOXELS]: The third parameter is the maximum number of voxels considered in the Y axis for each iteration.\n",
661 argv[0]);
662 return EXIT_FAILURE;
663 }
664 const char *const inputFileName = argv[1];
665 const char *outputFileName = NULL;
666 if (argc > 2) {
667 outputFileName = argv[2];
668 }
669
670 FILE *const inputFilePointer = fopen(inputFileName, "rb");
671 if (!inputFilePointer) {
672 fprintf(stderr, "Unable to open file '%s'!\n", inputFileName);
673 return EXIT_FAILURE;
674 }
675
676 if (readSetUP(inputFilePointer) == EXIT_FAILURE) {
677 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
678 return EXIT_FAILURE;
679 }
680 if (argc > 3) {
681 const int yMaxVoxels = atoi(argv[3]);
682 yVoxels = min(gl_nVoxel[Y], max(yMaxVoxels, 1));
683 } else {
684 yVoxels = min(gl_nVoxel[Y], yVoxels);
685 }
686 double partialTime = hpc_gettime();
687 // Number of angular positions
688 const unsigned short nTheta = gl_angularTrajectory / gl_positionsAngularDistance + 1;
689 const unsigned nSidePixels = gl_detectorSideLength / gl_pixelDim;
690 initEnvironment(nTheta);
691 // Array containing the coefficients of each voxel
692 double *const f = (double *) malloc(sizeof(double) * gl_nVoxel[X] * yVoxels * gl_nVoxel[Z]);
693 // Array containing the computed attenuation detected in each pixel of the detector
694 double *const g = (double *) calloc(nSidePixels * nSidePixels * nTheta, sizeof(double));
695 // double *const g = (double *) malloc(sizeof(double) * nSidePixels * nSidePixels * nTheta);
696 // Minimum and maximum attenuation computed
697 double gMinValue = INFINITY, gMaxValue = -INFINITY;
698 double totalTime = hpc_gettime() - partialTime;
699
700 // Iterates over object subsection
701 for (unsigned short slice = 0; slice < gl_nVoxel[Y]; slice += yVoxels) {
702 unsigned short nOfSlices;
703
704 if (gl_nVoxel[Y] - slice < yVoxels) {
705 nOfSlices = gl_nVoxel[Y] - slice;
706 } else {
707 nOfSlices = yVoxels;
708 }
709
710 // Read voxels coefficients
711 if (!fread(f, sizeof(double), (size_t) gl_nVoxel[X] * nOfSlices * gl_nVoxel[Z], inputFilePointer)) {
712 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
713 free(f);
714 free(g);
715 return EXIT_FAILURE;
716 }
717
718#ifdef PRINT
719 static unsigned short it = 0;
720 printf("IT %u of size %hu\n", ++it, nOfSlices);
721#endif
722 // Computes subsection projection
723 partialTime = hpc_gettime();
724 computeProjections(slice, f, g, &gMinValue, &gMaxValue, nTheta, nSidePixels);
725 totalTime += hpc_gettime() - partialTime;
726 }
727 fclose(inputFilePointer);
728 partialTime = hpc_gettime();
729 free(f);
730 termEnvironment();
731 totalTime += hpc_gettime() - partialTime;
732 printf("Execution time (s) %.2f\n", totalTime);
733
734 if (outputFileName != NULL) {
735 FILE *const outputFilePointer = fopen(outputFileName, "w");
736 if (!outputFileName) {
737 fprintf(stderr, "Unable to open file '%s'!\n", outputFileName);
738 free(g);
739 return EXIT_FAILURE;
740 }
741 // Iterates over each attenuation value computed, prints a value between [0-255]
742 fprintf(outputFilePointer, "P2\n%d %d\n255", nSidePixels, nSidePixels * nTheta);
743 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
744 double angle = -(double) gl_angularTrajectory / 2 + (double) positionIndex * gl_positionsAngularDistance;
745 fprintf(outputFilePointer, "\n#%lf", angle);
746 for (unsigned i = 0; i < nSidePixels; i++) {
747 fprintf(outputFilePointer, "\n");
748 for (unsigned j = 0; j < nSidePixels; j++) {
749 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + i * nSidePixels + j;
750 int color = (g[pixelIndex] - gMinValue) * 255 / (gMaxValue - gMinValue);
751 fprintf(outputFilePointer, "%d ", color);
752 }
753 }
754 }
755 fclose(outputFilePointer);
756 }
757 free(g);
758
759 return EXIT_SUCCESS;
760}
