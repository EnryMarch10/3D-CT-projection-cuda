218void getAllIntersections(const double source, const double pixel, const Ranges planeIndexesRanges, double *const a, const Axis axis)
219{
220 int start = 0, end = 0;
221 double d;
222
223 start = planeIndexesRanges.minIndx;
224 end = planeIndexesRanges.maxIndx;
225 if (end > start) {
226 double plane[end - start];
227 if (axis == X) {
228 plane[0] = getXPlane(start);
229 d = gl_voxelXDim;
230 if (pixel - source < 0) {
231 plane[0] = getXPlane(end);
232 d = -gl_voxelXDim;
233 }
234 } else if (axis == Y) {
235 plane[0] = getYPlane(start);
236 d = gl_voxelYDim;
237 if (pixel - source < 0) {
238 plane[0] = getYPlane(end);
239 d = -gl_voxelYDim;
240 }
241 } else /* if (axis == Z) */ {
242 plane[0] = getZPlane(start);
243 d = gl_voxelZDim;
244 if (pixel - source < 0) {
245 plane[0] = getZPlane(end);
246 d = -gl_voxelZDim;
247 }
248 }
249
250 for (unsigned short i = 1; i < end - start; i++) {
251 plane[i] = plane[i - 1] + d;
252 }
253 getIntersection(source, pixel, plane, end - start, a);
254 }

268Ranges getRangeOfIndex(const double source, const double pixel, const char isParallel, const double aMin, const double aMax, const Axis axis)
269{
270 Ranges idxs;
271 double firstPlane, lastPlane;
272 unsigned short voxelDim;
273
274 if (axis == X) {
275 voxelDim = gl_voxelXDim;
276 firstPlane = getXPlane(0);
277 lastPlane = getXPlane(gl_nPlanes[X] - 1);
278 } else if (axis == Y) {
279 voxelDim = gl_voxelYDim;
280 firstPlane = getYPlane(0);
281 lastPlane = getYPlane(gl_nPlanes[Y] - 1);
282 } else /* if (axis == Z) */ {
283 voxelDim = gl_voxelZDim;
284 firstPlane = getZPlane(0);
285 lastPlane = getZPlane(gl_nPlanes[Z] - 1);
286 }
287
288 // Gets range of indexes of XZ parallel planes
289 if (isParallel != Y) {
290 if (pixel - source >= 0) {
291 idxs.minIndx = gl_nPlanes[axis] - ceil((lastPlane - aMin * (pixel - source) - source) / voxelDim);
292 idxs.maxIndx = 1 + floor((aMax * (pixel - source) + source - firstPlane) / voxelDim);
293 } else {
294 idxs.minIndx = gl_nPlanes[axis] - ceil((lastPlane - aMax * (pixel - source) - source) / voxelDim);
295 idxs.maxIndx = floor((aMin * (pixel - source) + source - firstPlane) / voxelDim);
296 }
297 } else {
298 idxs.minIndx = 0;
299 idxs.maxIndx = 0;
300 }
301 return idxs;
302}

314int merge(const double *const a, const double *const b, const unsigned short lenA, const unsigned short lenB, double *const c)
315{
316 int i = 0, j = 0, k = 0;
317 while (j < lenA && k < lenB) {
318 if (a[j] < b[k]) {
319 c[i] = a[j];
320 j++;
321 } else {
322 c[i] = b[k];
323 k++;
324 }
325 i++;
326 }
327 while (j < lenA) {
328 c[i] = a[j];
329 i++;
330 j++;
331 }
332 while (k < lenB) {
333 c[i] = b[k];
334 i++;
335 k++;
336 }
337 return lenA + lenB;
338}

388Point getPixel(const double *const sinTable, const double *const cosTable, const unsigned r, const unsigned c, const unsigned short index)
389{
390 Point pixel;
391 const double sinAngle = sinTable[index];
392 const double cosAngle = cosTable[index];
393 const double elementOffset = gl_detectorSideLength / 2 - gl_pixelDim / 2;
394
395 pixel.x = -(double) gl_distanceObjectDetector * sinAngle + cosAngle * (-elementOffset + (double) gl_pixelDim * c);
396 pixel.y = -(double) gl_distanceObjectDetector * cosAngle - sinAngle * (-elementOffset + (double) gl_pixelDim * c);
397 pixel.z = -elementOffset + (double) gl_pixelDim * r;
398
399 return pixel;

451double computeAbsorption(const unsigned short slice, const Point source, const Point pixel, const double *const a, const unsigned short lenA, const double *const f)
452{
453 const double d12 = sqrt(pow(pixel.x - source.x, 2) + pow(pixel.y - source.y, 2) + pow(pixel.z - source.z, 2));
454 double g = 0.0;
455
456 if (lenA > 0) {
457 for (unsigned short i = 0; i < lenA - 1; i++) {
458 const double segments = d12 * (a[i + 1] - a[i]);
459 const double aMid = (a[i + 1] + a[i]) / 2;
460 const unsigned short xRow = min((source.x + aMid * (pixel.x - source.x) - getXPlane(0)) / gl_voxelXDim, gl_nVoxel[X] - 1);
461 const unsigned short yRow = min3((source.y + aMid * (pixel.y - source.y) - getYPlane(slice)) / gl_voxelYDim, gl_nVoxel[Y] - 1, OBJ_BUFFER - 1);
462 const unsigned short zRow = min((source.z + aMid * (pixel.z - source.z) - getZPlane(0)) / gl_voxelZDim, gl_nVoxel[Z] - 1);
463
464 g += f[(unsigned) yRow * gl_nVoxel[X] * gl_nVoxel[Z] + (unsigned) zRow * gl_nVoxel[Z] + xRow] * segments;
465 }
466 }
467 return g;
468}

481void computeProjections(const unsigned short slice, double *f, double *g, double *gMin, double *gMax, const unsigned short nTheta, const unsigned nSidePixels)
482{
483 double l_gMin = INFINITY;
484 double l_gMax = -INFINITY;
485
486 // Iterates over each source
487 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
488 const Point source = getSource(gl_sinTable, gl_cosTable, positionIndex);
489
490 // Iterates over each pixel of the detector
491#pragma omp parallel for collapse(2) schedule(dynamic) default(none) shared(gl_sinTable, gl_cosTable, nSidePixels, positionIndex, source, slice, f, g, nTheta, gl_nVoxel, gl_nPlanes) reduction(min:l_gMin) reduction(max:l_gMax)
492 for (unsigned r = 0; r < nSidePixels; r++) {
493 for (unsigned c = 0; c < nSidePixels; c++) {
494 double a[3][2];
495 double aMerged[gl_nPlanes[X] + gl_nPlanes[Y] + gl_nPlanes[Z]];
496 double aX[gl_nPlanes[X]];
497 double aY[gl_nPlanes[Y]];
498 double aZ[gl_nPlanes[Z]];
499
500 // Gets the pixel's center cartesian coordinates
501 const Point pixel = getPixel(gl_sinTable, gl_cosTable, r, c, positionIndex);
502
503 // Computes Min-Max parametric values
504 double aMin, aMax;
505 double sidesPlanes[2];
506 char isParallel = -1;
507 getSidesXPlanes(sidesPlanes);
508 if (!getIntersection(source.x, pixel.x, sidesPlanes, 2, &a[X][0])) {
509 isParallel = X;
510 }
511 getSidesYPlanes(sidesPlanes, slice);
512 if (!getIntersection(source.y, pixel.y, sidesPlanes, 2, &a[Y][0])) {
513 isParallel = Y;
514 }
515 getSidesZPlanes(sidesPlanes);
516 if (!getIntersection(source.z, pixel.z, sidesPlanes, 2, &a[Z][0])) {
517 isParallel = Z;
518 }
519
520 aMin = getAMin(a, isParallel);
521 aMax = getAMax(a, isParallel);
522
523 if (aMin < aMax) {
524 // Computes Min-Max plane indexes
525 Ranges indices[3];
526 indices[X] = getRangeOfIndex(source.x, pixel.x, isParallel, aMin, aMax, X);
527 indices[Y] = getRangeOfIndex(source.y, pixel.y, isParallel, aMin, aMax, Y);
528 indices[Z] = getRangeOfIndex(source.z, pixel.z, isParallel, aMin, aMax, Z);
529
530 // Computes lengths of the arrays containing parametric value of the intersection with each set of parallel planes
531 int lenX = indices[X].maxIndx - indices[X].minIndx;
532 int lenY = indices[Y].maxIndx - indices[Y].minIndx;
533 int lenZ = indices[Z].maxIndx - indices[Z].minIndx;
534 if (lenX < 0) {
535 lenX = 0;
536 }
537 if (lenY < 0) {
538 lenY = 0;
539 }
540 if (lenZ < 0) {
541 lenZ = 0;
542 }
543
544 // Computes ray-planes intersection Nx + Ny + Nz
545 getAllIntersections(source.x, pixel.x, indices[X], aX, X);
546 getAllIntersections(source.y, pixel.y, indices[Y], aY, Y);
547 getAllIntersections(source.z, pixel.z, indices[Z], aZ, Z);
548
549 // Computes segments Nx + Ny + Nz
550 const unsigned short lenA = merge3(aX, aY, aZ, lenX, lenY, lenZ, aMerged);
551
552 // Associates each segment to the respective voxel Nx + Ny + Nz
553 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + r * nSidePixels + c;
554 g[pixelIndex] += computeAbsorption(slice, source, pixel, aMerged, lenA, f);
555 l_gMax = fmax(l_gMax, g[pixelIndex]);
556 l_gMin = fmin(l_gMin, g[pixelIndex]);
557 }
558 }
559 }
560 }
561 *gMax = l_gMax;
562 *gMin = l_gMin;
563}

647int main(int argc, char *argv[])
648{
649 if (argc < 2 || argc > 3) {
650 fprintf(stderr, "Usage: %s INPUT [OUTPUT]\n"
651 "- INPUT: The first parameter is the name of the input file.\n"
652 "- [OUTPUT]: The second parameter is the name of a binary file to store the output at.\n",
653 argv[0]);
654 return EXIT_FAILURE;
655 }
656 const char *const inputFileName = argv[1];
657 const char *outputFileName = NULL;
658 if (argc > 2) {
659 outputFileName = argv[2];
660 }
661
662 FILE *const inputFilePointer = fopen(inputFileName, "rb");
663 if (!inputFilePointer) {
664 fprintf(stderr, "Unable to open file '%s'!\n", inputFileName);
665 return EXIT_FAILURE;
666 }
667
668 if (readSetUP(inputFilePointer) == EXIT_FAILURE) {
669 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
670 return EXIT_FAILURE;
671 }
672 double partialTime = hpc_gettime();
673 // Number of angular positions
674 const unsigned short nTheta = gl_angularTrajectory / gl_positionsAngularDistance + 1;
675 const unsigned nSidePixels = gl_detectorSideLength / gl_pixelDim;
676 initEnvironment(nTheta);
677 // Array containing the coefficients of each voxel
678 double *const f = (double *) malloc(sizeof(double) * gl_nVoxel[X] * gl_nVoxel[Z] * OBJ_BUFFER);
679 // Array containing the computed attenuation detected in each pixel of the detector
680 double *const g = (double *) calloc(nSidePixels * nSidePixels * nTheta, sizeof(double));
681 // double *const g = (double *) malloc(sizeof(double) * nSidePixels * nSidePixels * nTheta);
682 // Minimum and maximum attenuation computed
683 double gMinValue = INFINITY, gMaxValue = -INFINITY;
684 double totalTime = hpc_gettime() - partialTime;
685
686 // Iterates over object subsection
687 for (unsigned short slice = 0; slice < gl_nVoxel[Y]; slice += OBJ_BUFFER) {
688 unsigned short nOfSlices;
689
690 if (gl_nVoxel[Y] - slice < OBJ_BUFFER) {
691 nOfSlices = gl_nVoxel[Y] - slice;
692 } else {
693 nOfSlices = OBJ_BUFFER;
694 }
695
696 // Read voxels coefficients
697 if (!fread(f, sizeof(double), (size_t) gl_nVoxel[X] * gl_nVoxel[Z] * nOfSlices, inputFilePointer)) {
698 fprintf(stderr, "Unable to read from file '%s'!\n", inputFileName);
699 free(f);
700 free(g);
701 return EXIT_FAILURE;
702 }
703
704 // Computes subsection projection
705 partialTime = hpc_gettime();
706 computeProjections(slice, f, g, &gMinValue, &gMaxValue, nTheta, nSidePixels);
707 totalTime += hpc_gettime() - partialTime;
708 }
709 fclose(inputFilePointer);
710 partialTime = hpc_gettime();
711 free(f);
712 termEnvironment();
713 totalTime += hpc_gettime() - partialTime;
714 printf("Execution time (s) %.2f\n", totalTime);
715
716 if (outputFileName != NULL) {
717 FILE *const outputFilePointer = fopen(outputFileName, "w");
718 if (!outputFileName) {
719 fprintf(stderr, "Unable to open file '%s'!\n", outputFileName);
720 free(g);
721 return EXIT_FAILURE;
722 }
723 // Iterates over each attenuation value computed, prints a value between [0-255]
724 fprintf(outputFilePointer, "P2\n%d %d\n255", nSidePixels, nSidePixels * nTheta);
725 for (unsigned short positionIndex = 0; positionIndex < nTheta; positionIndex++) {
726 double angle = -(double) gl_angularTrajectory / 2 + (double) positionIndex * gl_positionsAngularDistance;
727 fprintf(outputFilePointer, "\n#%lf", angle);
728 for (unsigned i = 0; i < nSidePixels; i++) {
729 fprintf(outputFilePointer, "\n");
730 for (unsigned j = 0; j < nSidePixels; j++) {
731 const unsigned pixelIndex = positionIndex * nSidePixels * nSidePixels + i * nSidePixels + j;
732 int color = (g[pixelIndex] - gMinValue) * 255 / (gMaxValue - gMinValue);
733 fprintf(outputFilePointer, "%d ", color);
734 }
735 }
736 }
737 fclose(outputFilePointer);
738 }
739 free(g);
740
741 return EXIT_SUCCESS;
742}
