110void generateCubeSlice(double *f, int nOfSlices, int offset, int sideLength)
111{
112 const int innerToOuterDiff = gl_nVoxel[X] / 2 - sideLength / 2;
113 const int rightSide = innerToOuterDiff + sideLength;
114
115 // Iterates over each voxel of the grid
116#pragma omp parallel for collapse(3) default(none) shared(f, nOfSlices, gl_nVoxel, offset, sideLength, innerToOuterDiff, rightSide)
117 for (int n = 0 ; n < nOfSlices; n++) {
118 for (int i = 0; i < gl_nVoxel[Z]; i++) {
119 for (int j = 0; j < gl_nVoxel[X]; j++) {
120 if (i >= innerToOuterDiff && i <= rightSide && j >= innerToOuterDiff && j <= rightSide
121 && n + offset >= innerToOuterDiff && n + offset <= gl_nVoxel[Y] - innerToOuterDiff) {
122 // Voxel position is inside the cubic object
123 f[gl_nVoxel[Z] * i + j + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 1.0;
124 } else {
125 f[gl_nVoxel[Z] * i + j + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 0.0;
126 }
127 }
128 }
129 }

140void generateSphereSlice(double *f, int nOfSlices, int offset, int diameter)
141{
142 // Iterates over each voxel of the grid
143#pragma omp parallel for collapse(3) default(none) shared(f, nOfSlices, gl_nVoxel, offset, diameter, gl_objectSideLength, gl_voxelYDim, gl_voxelXDim, gl_voxelZDim)
144 for (int n = 0; n < nOfSlices; n++) {
145 for (int r = 0; r < gl_nVoxel[Z]; r++) {
146 for (int c = 0; c < gl_nVoxel[X]; c++) {
147 Point temp;
148 temp.y = -(gl_objectSideLength / 2) + (gl_voxelYDim / 2) + (n + offset) * gl_voxelYDim;
149 temp.x = -(gl_objectSideLength / 2) + (gl_voxelXDim / 2) + c * gl_voxelXDim;
150 temp.z = -(gl_objectSideLength / 2) + (gl_voxelZDim / 2) + r * gl_voxelZDim;
151 const double distance = sqrt(pow(temp.x, 2) + pow(temp.y, 2) + pow(temp.z, 2));
152 if (distance <= diameter && c < gl_nVoxel[Z] / 2) {
153 // Voxel position is inside the sphere object
154 f[gl_nVoxel[Z] * r + c + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 1.0;
155 } else {
156 f[gl_nVoxel[Z] * r + c + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 0.0;

171void generateCubeWithSphereSlice(double *f, int nOfSlices, int offset, const int sideLength)
172{
173 const int innerToOuterDiff = gl_nVoxel[X] / 2 - sideLength / 2;
174 const int rightSide = innerToOuterDiff + sideLength;
175 const Point sphereCenter = {-sideLength * gl_voxelXDim / 4, -sideLength * gl_voxelYDim / 4, -sideLength * gl_voxelZDim / 4};
176
177 // Iterates over each voxel of the grid
178#pragma omp parallel for collapse(3) default(none) shared(f, nOfSlices, offset, sideLength, gl_nVoxel, innerToOuterDiff, rightSide, sphereCenter, gl_objectSideLength, gl_voxelYDim, gl_voxelXDim, gl_voxelZDim)
179 for (int n = 0 ; n < nOfSlices; n++) {
180 for (int i = 0; i < gl_nVoxel[Z]; i++) {
181 for (int j = 0; j < gl_nVoxel[X]; j++) {
182 f[gl_nVoxel[Z] * i + j + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 0.0;
183 if ((i >= innerToOuterDiff) && (i <= rightSide) && (j >= innerToOuterDiff) && (j <= rightSide)
184 && (n + offset >= innerToOuterDiff) && (n + offset <= gl_nVoxel[Y] - innerToOuterDiff)) {
185 // Voxel position is inside the cubic object
186 Point temp;
187 temp.y = -(gl_objectSideLength / 2) + (gl_voxelYDim / 2) + (n + offset) * gl_voxelYDim;
188 temp.x = -(gl_objectSideLength / 2) + (gl_voxelXDim / 2) + j * gl_voxelXDim;
189 temp.z = -(gl_objectSideLength / 2) + (gl_voxelZDim / 2) + i * gl_voxelZDim;
190 const double distance = sqrt(pow(temp.x - sphereCenter.x, 2) + pow(temp.y - sphereCenter.y, 2) + pow(temp.z - sphereCenter.z, 2));
191
192 if (distance > sideLength * gl_voxelXDim / 6) {
193 // Voxel position is outside the spherical cavity
194 f[gl_nVoxel[Z] * i + j + n * gl_nVoxel[X] * gl_nVoxel[Z]] = 1.0;
195 }
196 }
197 }
198 }
199 }
